void setupTimeout() {
    if (!setup_found_on_1) {

void checkSetup() {
    // Read 1 coil for check for "alive"
    setStatus(RS485, {Idle_Addr, Read_Coils_Command, 0, 1, 0, 1, 0xcc16});
    // If no ack after 1 sec -> "dead"
    delayedCallMs(setupTimeout, 1000);
}

// ----- VERIFY NEW MODBUS ADDRESS -----
void verify_timeout() {
    if (setup_wait_verify) {
        setup_wait_verify = 0;

		#ifdef FLAG_SEND_PUSH
			setStatus(@APP_IDThatPress:32, {4, "Не удалось подтвердить новый адрес, попробуйте ещё раз"});
		#endif
    }
}

void verify_new_addr() {
    setup_wait_verify = 1;
    // Read 1 coil from new address
    setStatus(RS485, {NEWADDRESS, Read_Coils_Command, 0, 1, 0, 1, 0xcc16});
    // If no ack after 2 sec -> unlucky
    delayedCallMs(verify_timeout, 2000);
}

void onInit() {
	setup_found_on_1 = 0;
	cancelDelayedCall(setupTimeout);
	checkSetup();
}

// Function for forming string for push
void sendPushAddressChanged(u8 oldAddr, u8 newAddr)
{
    // Packing as {1, "str", 0}
	
	sprintf(pushText, "Адрес Modbus изменен: %d -> %d", oldAddr, newAddr);
    pushPacket[0] = 1;

    u8 j = 0;
    for (; j < (64 - 1); ++j) {
        u8 c = pushText[j];
        pushPacket[1 + j] = c;
        if (c == 0) break;
    }
    if (j == (64 - 1)) {
        pushPacket[1 + j] = 0;
    }

	#ifdef FLAG_SEND_PUSH
		setStatus(@APP_IDThatPress:32, pushPacket);
	#endif  
}

V-ID/RS485
{
    u8 ACnumInArray = findACbyAddress(opt(0));
	if (ACnumInArray == 0xFF) return;
	getStatus(@widgetAC_ID:@widgetAC_arrSubID[ACnumInArray], &ACWidgetData);
	
	//srvMessage("Data from AC %d(%d), %d, %d", opt(0), ACnumInArray, widgetAC_ID, widgetAC_arrSubID[ACnumInArray]);
	
	// === COILS (0x01): addr 0, count 15 ===
	if ( (opt(1) == Read_Coils_Command) && !flag )
	{
		// Read all modes and funcs
		if(lastAddr == 1) {
			setStatus(@widgetAC_ID:@widgetAC_arrSubID[ACnumInArray], { (ACWidgetData[0] & 0x0E) | (opt(3) & 0x01) } );	//
			setStatus(@widgetAC_ID:@widgetLed_arrSubID[ACnumInArray], 			(opt(3) >> 1) & 0x01);					//
			setStatus(@widgetAC_ID:@widgetBeeper_arrSubID[ACnumInArray], 		(opt(3) >> 4) & 0x01); 					//
			setStatus(@widgetAC_ID:@widgetAirHeat_arrSubID[ACnumInArray], 		(opt(4) >> 2) & 0x01);					//
			setStatus(@widgetAC_ID:@widgetResetFilter_arrSubID[ACnumInArray],  	(opt(4) >> 5) & 0x01); 					//
			setStatus(@widgetAC_ID:@widgetOutAirSupply_arrSubID[ACnumInArray],  (opt(4) >> 6) & 0x01); 					//
			setStatus(@widgetAC_ID:@widgetACconnected_arrSubID[ACnumInArray], 	(opt(5) >> 3) & 0x01); 					//
		}
	} 

	// === DISCRETE (0x02): addr 0, count 24 ===
	else if ((opt(1) == Read_Discrete_Command) && !flag) {
		u8 k;
        u8 b;
        u8 bit;
        u8 found = 0;
        u8 firstIdx = 0xFF;

        // looking for the 1st "true"
        for (k = 0; k < 25; ++k) {
            b  = opt(3 + (k >> 3));         // byte 
            bit = (b >> (k & 7)) & 0x01;    // bit
            if (bit) { 
				found = 1; 
				firstIdx = k; 
				break; 
			}
        }

        if (found) {
            u16 errNum = (u16)(firstIdx + 1);
            sprintf(errorMSG, "Error %u", errNum);
            // setStatus(AC1_ERRORCODE, &errorMSG);
            // setStatus(AC1_ERROR, 1);
			setStatus(@widgetAC_ID:@widgetErrorCode_arrSubID[ACnumInArray], &errorMSG);
    		setStatus(@widgetAC_ID:@widgetError_arrSubID[ACnumInArray], 1);
        } else {
            sprintf(errorMSG, "No error");
            // setStatus(AC1_ERRORCODE, &errorMSG);
            // setStatus(AC1_ERROR, 0);
			setStatus(@widgetAC_ID:@widgetErrorCode_arrSubID[ACnumInArray], &errorMSG);
    		setStatus(@widgetAC_ID:@widgetError_arrSubID[ACnumInArray], 0);
        }
	}

	// === HOLDING (0x03): addr 1, count 32 ===
	else if ( (opt(1) == Read_Holding_Command) && (opt(2) == (qntyArray[1] * 2)) && !flag )
	{
		if(lastAddr == 1) {
